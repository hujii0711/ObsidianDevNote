## 2. CLAUDE.md - 시스템의 두뇌

```markdown
# CLAUDE.md - 자율 개발 오케스트레이터

## 🎯 핵심 원칙
1. 개발자 입력 최소화 - 한 줄 요청으로 전체 파이프라인 실행
2. Lazy Loading - 필요한 문서만 그때그때 참조
3. Todo 추적 - 모든 진척을 TODO.md에 실시간 기록
4. 자가 복구 - 에러 발생 시 자동 수정 3회 시도
5. 실패 격리 - 어느 단계든 실패 시 즉시 중단 후 보고

## 📚 문서 참조 규칙 (Lazy Loading)

### 시작 시 항상 읽는 문서
- `docs/index.md` (목차)

### 트리거 기반 자동 참조
| 키워드 | 자동 로드 문서 |
|--------|--------------|
| 인증/로그인/회원 | docs/features/auth.md |
| 결제/환불/구독 | docs/features/payment.md |
| DB/스키마/마이그레이션 | docs/architecture/database.md |
| API/엔드포인트 | docs/architecture/api.md |
| 테스트/spec | docs/conventions/testing.md |
| 보안/취약점 | docs/conventions/security.md |

## 🤖 서브에이전트 실행 규칙

### 파이프라인 순서 (반드시 준수)
```

1. plan-agent → docs/agents/plan-agent.md 참조
2. code-agent → docs/agents/code-agent.md 참조
3. review-agent → docs/agents/review-agent.md 참조
4. test-agent → docs/agents/test-agent.md 참조
5. security-agent→ docs/agents/security-agent.md 참조

```

### 각 에이전트 실행 전 체크
- [ ] 이전 에이전트 결과 확인
- [ ] 필요 문서 Lazy Load
- [ ] TODO.md 상태 업데이트

### 자가 복구 규칙
- lint 에러: `npm run lint:fix` 자동 실행 (최대 3회)
- 타입 에러: 에러 메시지 분석 후 자동 수정
- 테스트 실패: 실패 원인 분석 후 코드 수정 1회 시도
- 보안 이슈: 자동 수정 금지 → 반드시 사용자 보고

## 📋 Todo 관리 규칙
- 작업 시작: `bash scripts/todo-manager.sh start "항목"`
- 작업 완료: `bash scripts/todo-manager.sh done "항목"`
- 작업 실패: `bash scripts/todo-manager.sh fail "항목" "이유"`
- 진척률 10% 변화마다 출력

## 🔒 절대 금지 사항
- 보안 에이전트 생략 금지
- 테스트 실패 상태로 PR 생성 금지
- 하드코딩된 시크릿 커밋 금지
- docs/ 전체 일괄 로딩 금지
```

## 3. 메인 오케스트레이터

```bash
#!/bin/bash
# scripts/orchestrator.sh

set -euo pipefail

TASK="${1:-}"
RESULTS_DIR="agent-results"
LOG_FILE="$RESULTS_DIR/pipeline.log"

# ── 초기화 ──────────────────────────────────────
init() {
  mkdir -p $RESULTS_DIR
  
  # 태스크 미입력 시 대화형 입력
  if [ -z "$TASK" ]; then
    echo "🤖 무엇을 개발할까요?"
    read -r TASK
  fi

  echo "🚀 파이프라인 시작: $TASK" | tee -a $LOG_FILE
  echo "시작: $(date)" | tee -a $LOG_FILE

  # GitHub 이슈 자동 생성
  ISSUE_URL=$(gh issue create \
    --title "$TASK" \
    --body "자동 생성된 이슈\n\n파이프라인: Plan→Code→Review→Test→Security" \
    --label "auto-pipeline")
  ISSUE_NUM=$(echo $ISSUE_URL | grep -o '[0-9]*$')
  echo "ISSUE_NUM=$ISSUE_NUM" >> $RESULTS_DIR/env.sh

  # Git Worktree 생성
  BRANCH="auto/issue-${ISSUE_NUM}-$(date +%s)"
  WORKTREE="../worktree-${ISSUE_NUM}"
  git worktree add -b "$BRANCH" "$WORKTREE" main
  echo "WORKTREE=$WORKTREE" >> $RESULTS_DIR/env.sh
  echo "BRANCH=$BRANCH" >> $RESULTS_DIR/env.sh

  # Lazy Loading: 태스크 키워드 기반 문서 로드
  bash scripts/doc-loader.sh "$TASK" > $RESULTS_DIR/context.md

  # TODO.md 초기화
  init_todo
}

# ── Todo 초기화 ──────────────────────────────────
init_todo() {
  cat > TODO.md << EOF
# 🤖 자율 파이프라인 진척 현황

## 태스크
> $TASK
> 이슈: #$ISSUE_NUM | 시작: $(date +%H:%M)

## 파이프라인
- [ ] 🗺️  Plan Agent
- [ ] 💻  Code Agent
- [ ] 🔍  Review Agent
- [ ] 🎭  Test Agent
- [ ] 🔒  Security Agent
- [ ] 🚀  PR 생성

## 세부 작업
- [ ] 문서 Lazy Load
- [ ] 플랜 생성
- [ ] Worktree 생성
- [ ] 코드 구현
- [ ] 린터 통과
- [ ] 타입 체크 통과
- [ ] 코드 리뷰 통과
- [ ] 유닛 테스트 통과
- [ ] E2E 테스트 통과
- [ ] 커버리지 80% 이상
- [ ] 보안 점검 통과
- [ ] PR 생성 완료
EOF
  echo "📋 TODO.md 초기화"
}

# ── 파이프라인 실행 ──────────────────────────────
run_pipeline() {
  source $RESULTS_DIR/env.sh

  # Worktree로 이동하여 실행
  cd $WORKTREE

  run_agent() {
    local NAME=$1
    local SCRIPT=$2
    
    echo ""
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo " $NAME 실행 중..."
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━"
    
    bash scripts/todo-manager.sh start "$NAME"
    
    if bash "scripts/agents/$SCRIPT" "$TASK"; then
      bash scripts/todo-manager.sh done "$NAME"
    else
      bash scripts/todo-manager.sh fail "$NAME" "에이전트 실패"
      
      # 실패 리포트 출력
      echo ""
      echo "❌ $NAME 실패 - 파이프라인 중단"
      cat $RESULTS_DIR/final-report.md 2>/dev/null || true
      exit 1
    fi
  }

  run_agent "🗺️  Plan Agent"     "plan-agent.sh"
  run_agent "💻  Code Agent"     "code-agent.sh"
  run_agent "🔍  Review Agent"   "review-agent.sh"
  run_agent "🎭  Test Agent"     "test-agent.sh"
  run_agent "🔒  Security Agent" "security-agent.sh"
}

# ── PR 생성 ──────────────────────────────────────
create_pr() {
  source $RESULTS_DIR/env.sh
  cd $WORKTREE

  bash scripts/todo-manager.sh start "🚀  PR 생성"

  git add -A
  git commit -m "feat: #${ISSUE_NUM} ${TASK}"
  git push origin "$BRANCH"

  gh pr create \
    --title "feat: #${ISSUE_NUM} ${TASK}" \
    --body "$(generate_pr_body)" \
    --assignee "@me"

  bash scripts/todo-manager.sh done "🚀  PR 생성"

  # Worktree 정리
  cd -
  git worktree remove "$WORKTREE"
}

generate_pr_body() {
  cat << EOF
## 📋 변경 사항
$(git log origin/main..HEAD --oneline)

## 🤖 파이프라인 결과
$(cat $RESULTS_DIR/final-report.md)

## ✅ 진척 현황
$(cat TODO.md)

Closes #${ISSUE_NUM}
EOF
}

# ── 메인 실행 ────────────────────────────────────
main() {
  init
  run_pipeline
  create_pr
  
  echo ""
  echo "🎉 완료! PR이 생성되었습니다."
  echo "$(gh pr list --limit 1)"
}

main
```

## 4. 에이전트 공통 베이스

```bash
# scripts/agents/common.sh

RESULTS_DIR="agent-results"
RETRY_MAX=3

# 자가 복구 포함 명령 실행
run_with_retry() {
  local CMD=$1
  local FIX_CMD=${2:-""}
  local COUNT=0

  while [ $COUNT -lt $RETRY_MAX ]; do
    if eval "$CMD"; then
      return 0
    fi

    COUNT=$((COUNT + 1))
    echo "⚠️  실패 ($COUNT/$RETRY_MAX) - 자동 수정 시도..."

    if [ -n "$FIX_CMD" ]; then
      eval "$FIX_CMD"
    fi
  done

  return 1
}

# 결과 저장
save_result() {
  local AGENT=$1
  local STATUS=$2
  local DATA=$3

  cat > "$RESULTS_DIR/${AGENT}.json" << EOF
{
  "agent": "$AGENT",
  "status": "$STATUS",
  "timestamp": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "data": $DATA
}
EOF
}

# 이전 에이전트 결과 읽기
read_result() {
  cat "$RESULTS_DIR/$1.json" 2>/dev/null || echo "{}"
}

# Claude API 호출
ask_claude() {
  local PROMPT=$1
  local CONTEXT=$(cat "$RESULTS_DIR/context.md" 2>/dev/null)

  claude --print "
컨텍스트:
$CONTEXT

$PROMPT
"
}
```

```bash
(plan.docs_needed || []).forEach(doc => {

"

save_result "plan-agent" "PASS" "$PLAN"
echo "✅ Plan Agent 완료"
```

```bash
npx playwright test \
  --reporter=json \
  2> $RESULTS_DIR/e2e-results.json

E2E_EXIT=$?

# 커버리지 확인
COVERAGE=$(node -e "
try {
  const summary = require('./coverage/coverage-summary.json');
  const lines = summary.total.lines.pct;
  console.log(lines);
} catch(e) { console.log(0); }
")

echo "📊 커버리지: $COVERAGE%"

# 실패 시 자동 수정 1회
if [ $UNIT_EXIT -ne 0 ]; then
  FAILURES=$(cat $RESULTS_DIR/unit-results.json | \
    node -e "
      const d=require('fs').readFileSync('/dev/stdin','utf8');
      const r=JSON.parse(d);
      console.log(JSON.stringify(r.testResults?.filter(t=>t.status==='failed')));
    ")
  
  ask_claude "테스트 실패를 수정하세요: $FAILURES"
  npm run test -- --coverage
  UNIT_EXIT=$?
fi

if [ $UNIT_EXIT -ne 0 ] || [ $E2E_EXIT -ne 0 ]; then
  save_result "test-agent" "FAIL" \
    "{\"coverage\": $COVERAGE, \"unit\": $UNIT_EXIT, \"e2e\": $E2E_EXIT}"
  exit 1
fi

if (( $(echo "$COVERAGE < 80" | bc -l) )); then
  save_result "test-agent" "FAIL" \
    "{\"reason\": \"커버리지 미달\", \"coverage\": $COVERAGE}"
  exit 1
fi

save_result "test-agent" "PASS" \
  "{\"coverage\": $COVERAGE}"
echo "✅ Test Agent 완료 (커버리지: $COVERAGE%)"
```

```bash
  } catch(e) { console.log(0); }
")

# 2. 하드코딩 시크릿 스캔
SECRETS=$(grep -rn \
  -E "(password|secret|api_key|token|private_key)\s*[=:]\s*['\"][^'\"]{8,}" \
  src/ 2>/dev/null || true)

# 3. Claude 보안 분석 (OWASP Top 10)
SECURITY=$(ask_claude "
보안 전문가로서 OWASP Top 10 기준으로 분석하세요:

$DIFF

JSON으로 응답:
{
  \"verdict\": \"PASS|FAIL\",
  \"critical\": [{\"type\": \"\", \"location\": \"\", \"description\": \"\"}],
  \"high\": [],
  \"medium\": [],
  \"recommendations\": []
}
JSON만 출력.
")

echo "$SECURITY" > $RESULTS_DIR/security.json

VERDICT=$(node -e "console.log(require('./$RESULTS_DIR/security.json').verdict)")

# 하드코딩 시크릿 발견 시 즉시 실패
if [ -n "$SECRETS" ]; then
  echo "🚨 하드코딩된 시크릿 발견!"
  echo "$SECRETS"
  save_result "security-agent" "FAIL" \
    "{\"reason\": \"하드코딩 시크릿\", \"details\": \"수동 확인 필요\"}"
  exit 1
fi

# Critical 취약점 발견 시 즉시 실패 (자동 수정 없음)
if [ "$VERDICT" = "FAIL" ] || [ "$AUDIT_CRITICAL" -gt 0 ]; then
  echo "🚨 보안 취약점 발견 - 수동 확인 필요"
  cat $RESULTS_DIR/security.json
  save_result "security-agent" "FAIL" "$SECURITY"
  exit 1
fi

save_result "security-agent" "PASS" "$SECURITY"
echo "✅ Security Agent 완료"
```

```javascript
// scripts/generate-report.js

const fs = require('fs');
```

## 7. GitHub Actions - CI 자동화

```yaml
# .github/workflows/auto-pipeline.yml
name: 자율 개발 파이프라인

on:
  issues:
    types: [labeled]   # 'auto-pipeline' 라벨 추가 시 실행

jobs:
  pipeline:
    if: contains(github.event.label.name, 'auto-pipeline')
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v4
      
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          
      - run: npm ci
      - run: npx playwright install --with-deps chromium
      
      - name: 자율 파이프라인 실행
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          bash scripts/orchestrator.sh "${{ github.event.issue.title }}"
          
      - name: 결과 업로드
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: pipeline-results
          path: |
            agent-results/
            playwright-report/
            TODO.md
```
